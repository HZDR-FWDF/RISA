126  TIFFSetField(tif.get(), TIFFTAG_IMAGEWIDTH, volume.width());
127  TIFFSetField(tif.get(), TIFFTAG_IMAGELENGTH, volume.height());
128  TIFFSetField(tif.get(), TIFFTAG_SAMPLESPERPIXEL, 1);
129  TIFFSetField(tif.get(), TIFFTAG_BITSPERSAMPLE,
130  detail::BitsPerSample<value_type>::value);

132  detail::SampleFormat<value_type>::value);
133 
134  auto slice = volume[i];
135  auto data = slice.data();
136  auto dataPtr = data;
137 
138  for (auto row = 0u; row < slice.height(); ++row) {
139  TIFFWriteScanline(tif.get(), dataPtr, row);
140  dataPtr += volume.width();
141  }
142

144  throw std::runtime_error {
145  "savers::TIFF: tiffio error while writing to " + path };
