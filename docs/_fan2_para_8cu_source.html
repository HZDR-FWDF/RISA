160 auto Fan2Para::process(input_type&& sinogram) -> void {
161  if (sinogram.valid()) {
162  BOOST_LOG_TRIVIAL(debug)<< "Fan2Para: Image arrived with Index: " << sinogram.index() << "to device " << sinogram.device();
163  fanSinograms_[sinogram.device()].push(std::move(sinogram));
164  } else {
165  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Fan2Para: Received sentinel, finishing.";
166 
167  //send sentinal to processor thread and wait 'til it's finished
168  for(auto i = 0; i < numberOfDevices_; i++) {
169  fanSinograms_[i].push(input_type());
170  }
171 
172  for(auto i = 0; i < numberOfDevices_; i++) {
173  processorThreads_[i].join();
174  }
175  //push sentinel to results for next stage
176  results_.push(output_type());
177  BOOST_LOG_TRIVIAL(info) << "recoLib::cuda::Fan2Para: Finished.";
178  }

189 auto Fan2Para::processor(const int deviceID) -> void {
190  //nvtxNameOsThreadA(pthread_self(), "Fan2Para");
191  CHECK(cudaSetDevice(deviceID));
192  dim3 blocks2D(blockSize2D_, blockSize2D_);
193  dim3 grids2D(
194  std::ceil(params_.numberOfParallelDetectors_ / (float) blockSize2D_),
195  std::ceil(
196  params_.numberOfParallelProjections_ / (float) blockSize2D_));
197  dim3 blocks1D(blockSize1D_);
198  dim3 grids1D(
199  std::ceil(
200  params_.numberOfParallelProjections_
201  * params_.numberOfParallelDetectors_
202  / (float) blockSize1D_));
203  auto params_d = glados::cuda::make_device_ptr<parameters>(1);
204  CHECK(
205  cudaMemcpyAsync(params_d.get(), &params_, sizeof(parameters),
206  cudaMemcpyHostToDevice, streams_[deviceID]));
207  CHECK(cudaStreamSynchronize(streams_[deviceID]));
208  CHECK(cudaFuncSetCacheConfig(interpolation, cudaFuncCachePreferL1));
209  BOOST_LOG_TRIVIAL(info)<< "recoLib::cuda::Fan2Para: Running Thread for Device " << deviceID;
210  while (true) {
211  auto sinogram = fanSinograms_[deviceID].take();
212  if (!sinogram.valid())
213  break;
214  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Fan2Para: Fan2Para of sinogram with Index " << sinogram.index();

216  //copy image from device to host
217  auto img = glados::MemoryPool<deviceManagerType>::instance()->requestMemory(
218  memoryPoolIdxs_[deviceID]);

220  setValue<float> <<<grids1D, blocks1D, 0, streams_[deviceID]>>>(
221  img.container().get(), 0.0, img.size());

226  gamma_d_[deviceID].get(), theta_d_[deviceID].get(),
227  alphaCircle_d_[deviceID].get(), s_d_[deviceID].get(),

229  thetaAfterRay2_d_[deviceID].get(),
230  thetaBeforeRay1_d_[deviceID].get(),
231  thetaBeforeRay2_d_[deviceID].get(),
232  gammaBeforeRay1_d_[deviceID].get(),
233  gammaBeforeRay2_d_[deviceID].get(),
234  gammaAfterRay1_d_[deviceID].get(),
235  gammaAfterRay2_d_[deviceID].get(), thetaGoalRay1_d_[deviceID].get(),
236  thetaGoalRay2_d_[deviceID].get(), gammaGoalRay1_d_[deviceID].get(),

251 auto Fan2Para::computeFan2ParaTransp() -> void {
252 
253  BOOST_LOG_TRIVIAL(info)<< "Computing Hash Table for conversion from fan to parallel beam.";
254 
255  auto dataSetSize = params_.numberOfParallelProjections_ * params_.numberOfParallelDetectors_
256  * params_.numberOfPlanes_;
257 
258  //allocate memory on host
259  theta_.resize(params_.numberOfFanProjections_);
260  gamma_.resize(params_.numberOfFanDetectors_);
261  s_.resize(params_.numberOfParallelDetectors_);
262  alphaCircle_.resize(params_.numberOfParallelProjections_);
263 
264  thetaAfterRay1_.resize(dataSetSize);
265  thetaAfterRay2_.resize(dataSetSize);
266  thetaBeforeRay1_.resize(dataSetSize);
267  thetaBeforeRay2_.resize(dataSetSize);
268  gammaAfterRay1_.resize(dataSetSize);
269  gammaAfterRay2_.resize(dataSetSize);
270  gammaBeforeRay1_.resize(dataSetSize);
271  gammaBeforeRay2_.resize(dataSetSize);
272  gammaGoalRay1_.resize(dataSetSize);
273  gammaGoalRay2_.resize(dataSetSize);
274  thetaGoalRay1_.resize(dataSetSize);
275  thetaGoalRay2_.resize(dataSetSize);
276  ray1_.resize(dataSetSize);
277  ray2_.resize(dataSetSize);
278 
279  // === Init values for Hash table
280  // ===============================
281  // == Theta = Ortswinkel des Quellpunktes auf Target
282  // == Gamma = Ortswinkel des Detektorpixels
283  // == s = diskreter Abstand der Detektorpixel (Para)
284  // == alpha_kreis = Ortswinkel der Parallelstrahlquellen
285 
286  for (auto j = 0; j < params_.numberOfFanProjections_; j++) {
287  theta_[j] = j * (360.0 / params_.numberOfFanProjections_) - params_.sourceOffset_;
288  if (theta_[j] < 0.0)
289  theta_[j] = theta_[j] + 360.0;
290  theta_[j] = ((2 * M_PI) / 360.0) * theta_[j];
291  }
292 
293  gamma_[0] = 0.0;
294  for (auto j = 1; j < params_.numberOfFanDetectors_; j++) {
295  gamma_[j] = j * ((360.0 / (float) params_.numberOfFanDetectors_));
296  gamma_[j] = ((2 * M_PI) / 360.0) * gamma_[j];
297  }
298 
299  for (auto j = 0; j < params_.numberOfParallelDetectors_; j++)
300  s_[j] = ((-0.5) * params_.imageWidth_)
301  + (((0.5 + j) * params_.imageWidth_) / (float) params_.numberOfParallelDetectors_);
302 
303  for (auto j = params_.numberOfParallelProjections_; j > 0; j--) {
304  alphaCircle_[j] = j * (360.0 / (float) params_.numberOfParallelProjections_);
305  alphaCircle_[j] = ((2 * M_PI) / 360.0) * alphaCircle_[j] + M_PI / 2;
306 
307  if (alphaCircle_[j] > 2 * M_PI)
308  alphaCircle_[j] = (alphaCircle_[j]) - (2.0 * M_PI);
309  }
310 
311  // === calculate Hash Table
312  // =========================
313  unsigned long long ind = 0;
314  int i, j, k;
315  float kappa = 0, L = 0;
316  float tb = 0.0;
317 
318  // Abfrage
319  if (params_.imageCenterY_ != 0) {
320  L = sqrt(params_.imageCenterY_ * params_.imageCenterY_ + params_.imageCenterX_ * params_.imageCenterX_);
321 
322  if (params_.imageCenterY_ < 0)
323  tb = 1.0;
324 
325  kappa = atan(params_.imageCenterX_ / params_.imageCenterY_) + tb * M_PI;
326  } else if (params_.imageCenterX_ != 0) {
327  L = sqrt(params_.imageCenterY_ * params_.imageCenterY_ + params_.imageCenterX_ * params_.imageCenterX_);
328 
329  if (params_.imageCenterX_ < 0)
330  kappa = -M_PI / 2.;
331  else
332  kappa = M_PI / 2.;
333  }
334 
335  unsigned int parallelSize = params_.numberOfParallelDetectors_
336  * params_.numberOfParallelProjections_;
337  float temp_1;
338 
339  for (k = 0; k < params_.numberOfPlanes_; k++) {
340  for (j = 0; j < params_.numberOfParallelProjections_; j++) {
341  for (i = 0; i < params_.numberOfParallelDetectors_; i++) {
342 
343  ind = j * params_.numberOfParallelDetectors_ + i + (k * parallelSize);
344  temp_1 = (s_[i] - L * sin(alphaCircle_[j] - kappa)) / params_.rDetector_;
345 
346  //Prüfen, ob asin möglich
347  if (temp_1 <= 1 || temp_1 >= -1)
348  computeAngles(i, j, ind, k, L, kappa);
349  }
350  }
351  }

354 auto Fan2Para::computeAngles(int i, int j, unsigned int ind, int k, float L,
355  float kappa) -> void {
356 
357  //Übergabe-Parameter
358  float epsilon = 0;
359 
360  //Hilfsvariable
361  float dif_best = M_PI, dif = 0, best_x = 0, temp_1 = 0, temp_2 = 0;
362  int x = 0;
363 
364  //Berechnungsvorschrift
365  //Theta
366  temp_1 = asin(((s_[i] - L * sin(alphaCircle_[j] - kappa)) / rTarget_[k])); //<-----Veränderung
367  thetaGoalRay1_[ind] = alphaCircle_[j] - temp_1;
368 
369  if (thetaGoalRay1_[ind] < 0)
370  thetaGoalRay1_[ind] = thetaGoalRay1_[ind] + 2.0 * M_PI;
371 
372  thetaGoalRay1_[ind] = ellipse_kreis_uwe(thetaGoalRay1_[ind], deltaX_[k],
373  deltaZ_[k], 2 * rTarget_[k]);
374 
375  thetaGoalRay2_[ind] = alphaCircle_[j] + temp_1 - M_PI;
376  if (thetaGoalRay2_[ind] < 0)
377  thetaGoalRay2_[ind] = thetaGoalRay2_[ind] + 2.0 * M_PI;
378 
379  thetaGoalRay2_[ind] = ellipse_kreis_uwe(thetaGoalRay2_[ind], deltaX_[k],
380  deltaZ_[k], 2 * rTarget_[k]);
381 
382  temp_1 = ((360.0 - sourceAngle_[k]) / 2.0) / 180.0 * M_PI;
383  temp_2 = (360.0 - ((360.0 - sourceAngle_[k]) / 2.0)) / 180.0 * M_PI;
384  if (thetaGoalRay1_[ind] > temp_1 && thetaGoalRay1_[ind] < temp_2)
385  ray1_[ind] = 1;
386  if (thetaGoalRay2_[ind] > temp_1 && thetaGoalRay2_[ind] < temp_2)
387  ray2_[ind] = 1;
388 
389  epsilon = asin(
390  ((s_[i] - L * sin(alphaCircle_[j] - kappa)) / params_.rDetector_)); //<-----Veränderung
391 
392  if (ray1_[ind]) {
393 
394  //Gamma
395  gammaGoalRay1_[ind] = epsilon + alphaCircle_[j] - 1.5 * M_PI;
396  if (gammaGoalRay1_[ind] < 0)
397  gammaGoalRay1_[ind] = gammaGoalRay1_[ind] + 2.0 * M_PI;
398  if (gammaGoalRay1_[ind] > 2 * M_PI)
399  gammaGoalRay1_[ind] = gammaGoalRay1_[ind] - 2.0 * M_PI;
400 
401  //Vektor Teta nach Wert durchsuchen für Fall 1
402  for (x = 0; x < params_.numberOfFanProjections_; x++) {
403  if (thetaGoalRay1_[ind] <= theta_[x]) {
404  dif = theta_[x] - thetaGoalRay1_[ind];
405  if (dif < dif_best) {
406  dif_best = dif;
407  best_x = x;
408  }
409  }
410  }
411 
412  if (best_x == 0) {
413  thetaBeforeRay1_[ind] = params_.numberOfFanProjections_ - 1;
414  thetaAfterRay1_[ind] = best_x;
415  } else {
416  thetaBeforeRay1_[ind] = best_x - 1;
417  thetaAfterRay1_[ind] = best_x;
418  }
419 
420  //Vektor Gamma nach Wert durchsuchen für Fall 1
421  for (x = 0; x < params_.numberOfFanDetectors_; x++) {
422  if (gammaGoalRay1_[ind] <= gamma_[x]) {
423  if (x == 0)
424  gammaBeforeRay1_[ind] = params_.numberOfFanDetectors_ - 1;
425  else
426  gammaBeforeRay1_[ind] = x - 1;
427  gammaAfterRay1_[ind] = x;
428  break;
429  }
430  }
431  if (gammaGoalRay1_[ind] > gamma_[params_.numberOfFanDetectors_ - 1]) {
432  gammaBeforeRay1_[ind] = params_.numberOfFanDetectors_ - 1;
433  gammaAfterRay1_[ind] = 0;
434  }
435  }
436 
437  if (ray2_[ind]) {
438  dif_best = M_PI;
439 
440  //Gamma für Fall 2
441  gammaGoalRay2_[ind] = -epsilon + alphaCircle_[j] - (M_PI / 2.0);
442  if (gammaGoalRay2_[ind] < 0)
443  gammaGoalRay2_[ind] = gammaGoalRay2_[ind] + 2.0 * M_PI;
444 
445  //Vektor Teta nach Wert durchsuchen für Fall 2
446  for (x = 0; x < params_.numberOfFanProjections_; x++) {
447  if (thetaGoalRay2_[ind] <= theta_[x]) {
448  dif = theta_[x] - thetaGoalRay2_[ind];
449  if (dif < dif_best) {
450  dif_best = dif;
451  best_x = x;
452  }
453  }
454  }
455  if (best_x == 0) {
456  thetaBeforeRay2_[ind] = params_.numberOfFanProjections_ - 1;
457  thetaAfterRay2_[ind] = best_x;
458 
459  } else {
460  thetaBeforeRay2_[ind] = best_x - 1;
461  thetaAfterRay2_[ind] = best_x;
462  }
463 
464  //Vektor Gamma nach Wert durchsuchen für Fall 2
465  for (x = 0; x < params_.numberOfFanDetectors_; x++) {
466  if (gammaGoalRay2_[ind] <= gamma_[x]) {
467  if (x == 0)
468  gammaBeforeRay2_[ind] = params_.numberOfFanDetectors_ - 1;
469  else
470  gammaBeforeRay2_[ind] = x - 1;
471  gammaAfterRay2_[ind] = x;
472  break;
473  }
474  }
475  if (gammaGoalRay2_[ind] > gamma_[params_.numberOfFanDetectors_ - 1]) {
476  gammaBeforeRay2_[ind] = params_.numberOfFanDetectors_ - 1;
477  gammaAfterRay2_[ind] = 0;
478  }
479  }
480 }
