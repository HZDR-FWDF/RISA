59 auto OfflineLoaderPerfTest::loadImage() -> glados::Image<manager_type> {
60  if (buffer_.empty())
61  return glados::Image<manager_type>();
62  auto sino = std::move(buffer_.front());
63  if (sino.index() > 0) {
64  tmr_.stop();
65  double duration = tmr_.elapsed();
66  if (duration < bestCaseTime_)
67  bestCaseTime_ = duration;
68  if (duration > worstCaseTime_)
69  worstCaseTime_ = duration;
70  }
71  buffer_.pop();
72  if (index_ < stopFrame_) {
73  auto img = glados::MemoryPool<manager_type>::instance()->requestMemory(
74  memoryPoolIndex_);
75  img.setIdx(index_);
76  buffer_.push(std::move(img));
77  index_++;
78  }
79  tmr_.start();
80  sino.setStart(std::chrono::high_resolution_clock::now());
81  return sino;
82 }

84 auto OfflineLoaderPerfTest::readInput() -> void {
85  Timer tmr1, tmr2;
86  std::vector<std::vector<unsigned short>> fileContents(
87  numberOfDetectorModules_);
88  int numberOfDetPerModule = numberOfDetectors_ / numberOfDetectorModules_;
89  if (path_.back() != '/')
90  path_.append("/");
91  tmr1.start();
92  tmr2.start();
93 #pragma omp parallel for default(shared) num_threads(numberOfDetectorModules_/3)
94  for (auto i = 1; i <= numberOfDetectorModules_; i++) {
95  std::vector<unsigned short> content;
96  std::ifstream input(path_ + fileName_ + std::to_string(i) + fileEnding_,
97  std::ios::in | std::ios::binary);
98  if (!input) {
99  BOOST_LOG_TRIVIAL(error)<< "recoLib::OfflineLoader: Source file could not be loaded.";
100  throw std::runtime_error("File could not be opened. Please check!");
101  }
102  //allocate memory in vector
103  std::streampos fileSize;
104  input.seekg(0, std::ios::end);
105  fileSize = input.tellg();
106  input.seekg(0, std::ios::beg);
107  content.resize(fileSize / sizeof(unsigned short));
108  input.read((char*) &content[0], fileSize);
109  fileContents[i - 1] = content;
110  }
111  tmr2.stop();
112  for (unsigned int i = 0; i < numberOfFrames_; i++) {
113  for (auto planeInd = 0; planeInd < numberOfPlanes_; planeInd++) {
114  auto sino = glados::MemoryPool<manager_type>::instance()->requestMemory(memoryPoolIndex_);
115  for (auto detModInd = 0; detModInd < numberOfDetectorModules_;
116  detModInd++) {
117  std::size_t startIndex = (planeInd + i * numberOfPlanes_) * numberOfDetPerModule * numberOfProjections_;
118  std::copy(fileContents[detModInd].cbegin() + startIndex, fileContents[detModInd].cbegin() + startIndex
119  + numberOfDetPerModule * numberOfProjections_, sino.container().get()
120  + detModInd * numberOfDetPerModule * numberOfProjections_);
121  }
122  sino.setIdx(planeInd + i * numberOfPlanes_);
123  sino.setPlane(planeInd);
124  buffer_.push(std::move(sino));
125  }
126  }
127  tmr1.stop();
128  double totalFileSize = numberOfProjections_ * numberOfDetectors_
129  * numberOfPlanes_ * numberOfFrames_ * sizeof(unsigned short) / 1024.0
130  / 1024.0;
131  BOOST_LOG_TRIVIAL(info)<< "recoLib::OfllineLoader: Reading and sorting data input took " << tmr1.elapsed() << " s, " << totalFileSize/tmr2.elapsed() << " MByte/s.";
132 }

142 auto OfflineLoaderPerfTest::readConfig(const std::string& configFile) -> bool {
143  ConfigReader configReader = ConfigReader(configFile.data());
144  int samplingRate, scanRate;
145  if (configReader.lookupValue("numberOfFanDetectors", numberOfDetectors_)
146  && configReader.lookupValue("numberOfDetectorModules",
147  numberOfDetectorModules_)
148  && configReader.lookupValue("dataInputPath", path_)
149  && configReader.lookupValue("dataFileName", fileName_)
150  && configReader.lookupValue("dataFileEnding", fileEnding_)
151  && configReader.lookupValue("numberOfPlanes", numberOfPlanes_)
152  && configReader.lookupValue("samplingRate", samplingRate)
153  && configReader.lookupValue("scanRate", scanRate)
154  && configReader.lookupValue("numberOfDataFrames", numberOfFrames_)) {
155  numberOfProjections_ = samplingRate * 1000000 / scanRate;
156  return EXIT_SUCCESS;
157  }
158 
159  return EXIT_FAILURE;
160 }
