88 auto Masking::process(input_type&& img) -> void {
89  if (img.valid()) {
90  BOOST_LOG_TRIVIAL(debug)<< "CropImage: Image arrived with Index: " << img.index() << "to device " << img.device();
91  imgs_[img.device()].push(std::move(img));
92  } else {
93  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::CropImage: Received sentinel, finishing.";
94 
95  //send sentinal to processor thread and wait 'til it's finished
96  for(auto i = 0; i < numberOfDevices_; i++) {
97  imgs_[i].push(input_type());
98  }
99 
100  for(auto i = 0; i < numberOfDevices_; i++) {
101  processorThreads_[i].join();
102  }
103  //push sentinel to results for next stage
104  results_.push(output_type());
105  BOOST_LOG_TRIVIAL(info) << "recoLib::cuda::CropImage: Finished.";
106  }

113 auto Masking::processor(const int deviceID) -> void {
114  //nvtxNameOsThreadA(pthread_self(), "CropImage");
115  CHECK(cudaSetDevice(deviceID));
116  dim3 blocks(16, 16);
117  dim3 grids(std::ceil(numberOfPixels_ / 16.0),
118  std::ceil(numberOfPixels_ / 16.0));
119  BOOST_LOG_TRIVIAL(info)<< "recoLib::cuda::CropImage: Running Thread for Device " << deviceID;
120  while (true) {
121  auto img = imgs_[deviceID].take();
122  if (!img.valid())
123  break;
124  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::CropImage: CropImageing image with Index " << img.index();
125 
126  //normalization
127  if(performNormalization_){
128  auto pair = (thrust::minmax_element(thrust::device_pointer_cast(img.container().get()), thrust::device_pointer_cast(img.container().get()+img.size())));
129  float min = *pair.first;
130  float max = *pair.second;
131  float diff = max- min;
132  thrust::transform(thrust::device_pointer_cast(img.container().get()), thrust::device_pointer_cast(img.container().get()+img.size()), thrust::device_pointer_cast(img.container().get()), (thrust::placeholders::_1 - min)/diff);
133  }
134  mask<<<grids, blocks, 0, streams_[deviceID]>>>(img.container().get(),
135  maskingValue_ ,numberOfPixels_);
136  CHECK(cudaPeekAtLastError());
137 
138  //wait until work on device is finished
139  CHECK(cudaStreamSynchronize(streams_[deviceID]));
140  results_.push(std::move(img));
141 
142  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::CropImage: CropImageing image with Index " << img.index() << " finished.";
143  }

157 __global__ void mask(float* __restrict__ img, const float value, const int numberOfPixels) {
158  const auto x = glados::cuda::getX();
159  const auto y = glados::cuda::getY();
160  if (x >= numberOfPixels || y >= numberOfPixels)
161  return;
162  const float center = (numberOfPixels - 1.0) * 0.5;
163  const float dX = x - center;
164  const float dY = y - center;
165  const float distance = dX * dX + dY * dY;
166  if (distance > numberOfPixels * numberOfPixels * 0.25) {
167  img[x + numberOfPixels * y] = value;
168  }
169 }

171 }
