56 auto ReceiverModule::run() -> void {
57  std::size_t headerSize{(sizeof(std::size_t)+sizeof(unsigned short))/sizeof(unsigned short)};
58  std::vector<unsigned short> buf(numberOfProjectionsPerPacket_*numberOfDetectorsPerModule_ + headerSize);
59  unsigned int sinoSize = numberOfProjections_*numberOfDetectorsPerModule_;
60  int numBytes = 0;
61  unsigned short numberOfParts = numberOfProjections_/numberOfProjectionsPerPacket_ - 1;
62  if(transportProtocol_ == transportProtocol::TCP){
63  boost::asio::io_service io_service;
64 
65  tcp::acceptor acceptor(io_service, tcp::endpoint(tcp::v4(), port_));
66 
67  tcp::socket socket(io_service);
68  acceptor.accept(socket);
69 
70  boost::system::error_code error;
71 
72  while(true){
73  boost::asio::read(socket, boost::asio::buffer(buf.data(), buf.size()*sizeof(unsigned short)), error);
74  if (error == boost::asio::error::eof)
75  break; // Connection closed cleanly by peer.
76  std::size_t index = *((std::size_t *)buf.data());
77  unsigned short partID = *((unsigned short*)(buf.data() + sizeof(std::size_t)/sizeof(unsigned short)));
78  BOOST_LOG_TRIVIAL(debug) << "ReceiverModule " << moduleID_ << " received packet " << index << " part ID: " << partID;
79  std::copy(buf.cbegin() + headerSize, buf.cend(), buffer_.begin() + sinoSize * (index%bufferSize_) + partID * numberOfProjectionsPerPacket_*numberOfDetectorsPerModule_);
80  if(numberOfParts == partID)
81  notification_.notify(moduleID_, index);
82  }
83  }else if(transportProtocol_ == transportProtocol::UDP){
84  //Possibility how to realize timeout with boost::asio::udp was not found yet
85  //perhaps a final packet needs to be send, that specifies the end of transaction
86  /*boost::asio::io_service io_service;
87  udp::socket socket(io_service);
88  udp::endpoint listen_endpoint(boost::asio::ip::address::from_string(address_), port_);
89  socket.open(listen_endpoint.protocol());
90  socket.bind(listen_endpoint);*/
91  while(true){
92  numBytes = udpServer_.timed_recv((char*)(buf.data()), buf.size()*sizeof(unsigned short), timeout_);
93  if(numBytes < 0) break;
94  BOOST_LOG_TRIVIAL(debug) << "Number of bytes received: " << numBytes;
95  /*boost::system::error_code ec;
96  std::size_t n = socket.receive_from(boost::asio::buffer(buf.data(), buf.size()*sizeof(unsigned short)), listen_endpoint);
97  */
98  std::size_t index = *((std::size_t *)buf.data());
99  unsigned short partID = *((unsigned short*)(buf.data() + sizeof(std::size_t)/sizeof(unsigned short)));
100  int diff = index*numberOfParts + partID - lastIndex_;
101  if(diff > 1){
102  BOOST_LOG_TRIVIAL(warning) << "ReceiverModule " << moduleID_ << ": Lost package or wrong order. Last " << lastIndex_ << " new: " << index*numberOfParts + partID;
103  lastIndex_ = index*numberOfParts + partID;
104  //continue;
105  }
106  lastIndex_ = index*numberOfParts + partID;
107  BOOST_LOG_TRIVIAL(debug) << "ReceiverModule " << moduleID_ << " received packet " << index << " partID: " << partID;
108  auto it = std::count(buf.cbegin()+headerSize, buf.cend(), 0);
109  std::copy(buf.cbegin() + headerSize, buf.cend(), buffer_.begin() + sinoSize * (index%bufferSize_) + partID * numberOfProjectionsPerPacket_*numberOfDetectorsPerModule_);
110  if(numberOfParts == partID)
111  notification_.notify(moduleID_, index);
112  }
113  }
114  notification_.notify(moduleID_, -1);
115  BOOST_LOG_TRIVIAL(info) << "ReceiverModul " << moduleID_ << ": No packets arriving since " << timeout_ << "s. Finishing.";
116 }

118 auto ReceiverModule::readConfig(const std::string& configFile) -> bool {
119  ConfigReader configReader = ConfigReader(configFile.data());
120  int samplingRate, scanRate;
121  std::string transportProt;
122  if (configReader.lookupValue("samplingRate", samplingRate)
123  && configReader.lookupValue("numberOfFanDetectors", numberOfDetectors_)
124  && configReader.lookupValue("scanRate", scanRate)
125  && configReader.lookupValue("transportProtocol", transportProt)
126  && configReader.lookupValue("timeout", timeout_)
127  && configReader.lookupValue("numberOfDetectorsPerModule", numberOfDetectorsPerModule_)
128  && configReader.lookupValue("numberOfProjectionsPerPacket", numberOfProjectionsPerPacket_)
129  && configReader.lookupValue("inputBufferSize", bufferSize_)
130  && configReader.lookupValue("numberOfDetectorModules", numberOfDetectorModules_)) {
131  if(transportProt == "udp")
132  transportProtocol_ = transportProtocol::UDP;
133  else if(transportProt == "tcp")
134  transportProtocol_ = transportProtocol::TCP;
135  else
136  transportProtocol_ = transportProtocol::UDP;
137  numberOfProjections_ = samplingRate * 1000000 / scanRate;
138  return EXIT_SUCCESS;
139  }
140 
141  return EXIT_FAILURE;
142 }
