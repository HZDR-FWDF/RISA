70  {
71  using empty_return = Image<MemoryManager>;

73  auto tif = std::unique_ptr<::TIFF, detail::TIFFDeleter>{TIFFOpen(path.c_str(), "rb")};
74  BOOST_LOG_TRIVIAL(debug) << "glados::loaders::TIFF: Open file " << path << " for reading.";
75  if(tif == nullptr)
76  throw std::runtime_error{"glados::loaders::TIFF: Could not open file " + path + " for reading."};
77 
78  int imageWidth, imageLength;
79 
80  TIFFGetField(tif.get(), TIFFTAG_IMAGELENGTH, &imageLength);
81  TIFFGetField(tif.get(), TIFFTAG_IMAGEWIDTH, &imageWidth);
82  // read image data
83  auto img = MemoryPool<MemoryManager>::instance()->requestMemory(memoryPoolIndex_);
84 
85  for(auto row = 0; row < imageLength; row++){
86  if(TIFFReadScanline(tif.get(), img.container().get() + row * imageWidth, row, 0) != 1){
87  throw std::runtime_error{"glados::loaders::TIFF: Could not read scanline."};
88  }
89  }
90  img.setIdx(index);
91  img.setPlane(0);
92 
93  BOOST_LOG_TRIVIAL(debug) << "glados::loaders::TIFF: Sent file " << path << ".";
94 
95  return std::move(img);
96  }
