79 UDPServer::UDPServer(const std::string& addr, int port)
80  : f_port(port)
81  , f_addr(addr)
82 {
83  char decimal_port[16];
84  snprintf(decimal_port, sizeof(decimal_port), "%d", f_port);
85  decimal_port[sizeof(decimal_port) / sizeof(decimal_port[0]) - 1] = '\0';
86  struct addrinfo hints;
87  memset(&hints, 0, sizeof(hints));
88  hints.ai_family = AF_UNSPEC;
89  hints.ai_socktype = SOCK_DGRAM;
90  hints.ai_protocol = IPPROTO_UDP;
91  int r(getaddrinfo(addr.c_str(), decimal_port, &hints, &f_addrinfo));
92  if(r != 0 || f_addrinfo == NULL)
93  {
94  throw udp_client_server_runtime_error(("invalid address or port for UDP socket: \"" + addr + ":" + decimal_port + "\"").c_str());
95  }
96  f_socket = socket(f_addrinfo->ai_family, SOCK_DGRAM | SOCK_CLOEXEC, IPPROTO_UDP);
97  if(f_socket == -1)
98  {
99  freeaddrinfo(f_addrinfo);
100  throw udp_client_server_runtime_error(("could not create UDP socket for: \"" + addr + ":" + decimal_port + "\"").c_str());
101  }
102  r = bind(f_socket, f_addrinfo->ai_addr, f_addrinfo->ai_addrlen);
103  if(r != 0)
104  {
105  freeaddrinfo(f_addrinfo);
106  close(f_socket);
107  throw udp_client_server_runtime_error(("could not bind UDP socket with: \"" + addr + ":" + decimal_port + "\"").c_str());
108  }
109 }
