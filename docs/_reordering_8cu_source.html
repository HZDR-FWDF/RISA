85 auto Reordering::process(input_type&& img) -> void {
86  if (img.valid()) {
87  BOOST_LOG_TRIVIAL(debug)<< "CropImage: Image arrived with Index: " << img.index() << "to device " << img.device();
88  sinos_[img.device()].push(std::move(img));
89  } else {
90  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::CropImage: Received sentinel, finishing.";
91 
92  //send sentinal to processor thread and wait 'til it's finished
93  for(auto i = 0; i < numberOfDevices_; i++) {
94  sinos_[i].push(input_type());
95  }
96 
97  for(auto i = 0; i < numberOfDevices_; i++) {
98  processorThreads_[i].join();
99  }
100  //push sentinel to results for next stage
101  results_.push(output_type());
102  BOOST_LOG_TRIVIAL(info) << "recoLib::cuda::CropImage: Finished.";
103  }
104 }

118 auto Reordering::processor(const int deviceID) -> void {
119  //nvtxNameOsThreadA(pthread_self(), "CropImage");
120  CHECK(cudaSetDevice(deviceID));
121  dim3 blocks(16, 16);
122  dim3 grids(std::ceil(numberOfFanDetectors_/16.0),
123  std::ceil(numberOfFanProjections_/16.0));
124 
125  std::vector<int> hashTable(numberOfFanDetectors_*numberOfFanProjections_);
126  createHashTable(hashTable);
127 
128  auto d_hashTable = glados::cuda::make_device_ptr<int>(numberOfFanDetectors_*numberOfFanProjections_);
129  CHECK(cudaMemcpy(d_hashTable.get(), hashTable.data(), sizeof(int)*hashTable.size(), cudaMemcpyHostToDevice));
130 
131  BOOST_LOG_TRIVIAL(info)<< "recoLib::cuda::Reordering: Running Thread for Device " << deviceID;
132  while (true) {
133  auto img = sinos_[deviceID].take();
134  if (!img.valid())
135  break;
136  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Reordering: Reordering image with Index " << img.index();
137 
138  auto sino_ordered = glados::MemoryPool<deviceManagerType>::instance()->requestMemory(memoryPoolIdxs_[deviceID]);
139 
140  reorder<<<grids, blocks, 0, streams_[deviceID]>>>(img.container().get(), sino_ordered.container().get(), d_hashTable.get(), numberOfFanProjections_, numberOfFanDetectors_);
141  CHECK(cudaPeekAtLastError());
142 
143  sino_ordered.setIdx(img.index());
144  sino_ordered.setDevice(img.device());
145  sino_ordered.setPlane(img.plane());
146  sino_ordered.setStart(img.start());
147 
148  //wait until work on device is finished
149  CHECK(cudaStreamSynchronize(streams_[deviceID]));
150  results_.push(std::move(sino_ordered));
151 
152  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Reordering: Reordering image with Index " << img.index() << " finished.";
153  }
154 }

164 auto Reordering::readConfig(const std::string& configFile) -> bool {
165  ConfigReader configReader = ConfigReader(configFile.data());
166  int samplingRate, scanRate;
167  if (configReader.lookupValue("numberOfFanDetectors", numberOfFanDetectors_)
168  && configReader.lookupValue("memPoolSize_Reordering", memPoolSize_)
169  && configReader.lookupValue("samplingRate", samplingRate)
170  && configReader.lookupValue("scanRate", scanRate)){
171  numberOfDetectorsPerModule_ = 16;
172  numberOfFanProjections_ = samplingRate * 1000000 / scanRate;
173  return EXIT_SUCCESS;
174  }
175  else
176  return EXIT_FAILURE;
177 
178 }

180 auto Reordering::createHashTable(std::vector<int>& hashTable) -> void {
181  int numberOfModules = 27;
182  int i = 0;
183  hashTable.resize(numberOfFanProjections_*numberOfFanDetectors_);
184  for(auto projInd = 0; projInd < numberOfFanProjections_; projInd++){
185  for(auto modInd = 0; modInd < numberOfModules; modInd++){
186  for(auto detInd = 0; detInd < numberOfDetectorsPerModule_; detInd++){
187  int index = detInd + projInd * numberOfDetectorsPerModule_ + modInd * numberOfDetectorsPerModule_*numberOfFanProjections_;
188  hashTable[i] = index;
189  i++;
190  }
191  }
192  }

195 __global__ void reorder(const unsigned short* __restrict__ unorderedSino, unsigned short* __restrict__ orderedSino,
196  const int* __restrict__ hashTable, const int numberOfProjections, const int numberOfDetectors) {
197  const auto x = glados::cuda::getX();
198  const auto y = glados::cuda::getY();
199  if (x >= numberOfDetectors || y >= numberOfProjections)
200  return;
201 
202  const int index = x + y * numberOfDetectors;
203  orderedSino[index] = unorderedSino[hashTable[index]];
204 }

206 }
