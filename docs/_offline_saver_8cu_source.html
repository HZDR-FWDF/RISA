102 auto OfflineSaver::saveImage(glados::Image<manager_type> image,
103  std::string path) -> void {
104  auto img = glados::MemoryPool<manager_type>::instance()->requestMemory(
105  memoryPoolIndex_);
106  std::copy(image.container().get(),
107  image.container().get() + image.size(),
108  img.container().get());
109  img.setIdx(image.index());
110  img.setPlane(image.plane());
111  img.setStart(image.start());
112  auto latency = img.duration();
113  if(latency < minLatency_)
114  minLatency_ = latency;
115  else if(latency > maxLatency_)
116  maxLatency_ = latency;
117  outputBuffers_[image.plane()].push_back(std::move(img));
118  if(mode_ == detail::RecoMode::offline){
119  if(outputBuffers_[image.plane()].full()){
120  writeTiffSequence(image.plane());
121  }
122  }else if(mode_ == detail::RecoMode::online){
123  tmrs_[image.plane()].stop();
124  double diff = tmrs_[image.plane()].elapsed();
125  if(diff < 0.02)
126  return;
127  std::string path = outputPath_ + fileName_ + "_plane"
128  + std::to_string(image.plane()) + "_" + "0" + ".tif";
129  auto tif = std::unique_ptr<::TIFF, detail::TIFFDeleter> { TIFFOpen(
130  path.c_str(), "wb") };
131  if (tif == nullptr)
132  throw std::runtime_error { "recoLib::OfflineSaver: Could not open file "
133  + path + " for writing." };
134  writeToTiff(tif.get(), std::move(image));
135  tmrs_[image.plane()].start();
136  }
137 }

139 auto OfflineSaver::writeTiffSequence(const int planeID) -> void {
140  if(outputBuffers_[planeID].count() == 0)
141  return;
142  std::string path = outputPath_ + fileName_ + "_plane"
143  + std::to_string(planeID) + "_" + std::to_string(fileIndex_[planeID]) + "seq.tif";
144  auto tif = std::unique_ptr<::TIFF, detail::TIFFDeleter> { TIFFOpen(
145  path.c_str(), "w8") };
146  if (tif == nullptr)
147  throw std::runtime_error { "recoLib::OfflineSaver: Could not open file "
148  + path + " for writing." };
149 
150  for(auto i = 0u; i < outputBuffers_[planeID].count(); i++){
151  BOOST_LOG_TRIVIAL(debug) << "writing out index " << i;
152  writeToTiff(tif.get(), std::move(outputBuffers_[planeID].at(i)));
153  if(TIFFWriteDirectory(tif.get()) != 1)
154  throw std::runtime_error{"savers::TIFF: tiffio error while writing to " + path};
155  }
156  ++fileIndex_[planeID];
157  outputBuffers_[planeID].clear();
158 }

160 auto OfflineSaver::writeToTiff(::TIFF* tif, glados::Image<manager_type> img) const -> void {
161 
162  TIFFSetField(tif, TIFFTAG_IMAGEWIDTH, numberOfPixels_);
163  TIFFSetField(tif, TIFFTAG_IMAGELENGTH, numberOfPixels_);
164  TIFFSetField(tif, TIFFTAG_BITSPERSAMPLE, 32);
165  TIFFSetField(tif, TIFFTAG_COMPRESSION, COMPRESSION_NONE);
166  TIFFSetField(tif, TIFFTAG_PHOTOMETRIC, PHOTOMETRIC_MINISBLACK);
167  TIFFSetField(tif, TIFFTAG_THRESHHOLDING, THRESHHOLD_BILEVEL);
168  TIFFSetField(tif, TIFFTAG_SAMPLESPERPIXEL, 1);
169  TIFFSetField(tif, TIFFTAG_SOFTWARE, "RISA");
170  TIFFSetField(tif, TIFFTAG_SAMPLEFORMAT, SAMPLEFORMAT_IEEEFP);
171 
172  auto data = img.data();
173  auto dataPtr = data;
174  for (auto row = 0; row < numberOfPixels_; ++row) {
175  TIFFWriteScanline(tif, dataPtr, row);
176  dataPtr += numberOfPixels_;
177  }
178 }

188 auto OfflineSaver::readConfig(const std::string& configFile) -> bool {
189  ConfigReader configReader = ConfigReader(configFile.data());
190  std::string mode;
191  if (configReader.lookupValue("numberOfPixels", numberOfPixels_)
192  && configReader.lookupValue("outputPath", outputPath_)
193  && configReader.lookupValue("outputFileName", fileName_)
194  && configReader.lookupValue("numberOfDataFrames", numberOfFrames_)
195  && configReader.lookupValue("numberOfPlanes", numberOfPlanes_)
196  && configReader.lookupValue("outputBufferSize", circularBufferSize_)
197  && configReader.lookupValue("mode", mode)) {
198  if(mode == "offline"){
199  mode_ = detail::RecoMode::offline;
200  }else if(mode == "online"){
201  mode_ = detail::RecoMode::online;
202  }else{
203  BOOST_LOG_TRIVIAL(error) << "Requested mode \"" << mode << "\" not supported.";
204  return EXIT_FAILURE;
205  }
206  return EXIT_SUCCESS;
207  }
208 
209  return EXIT_FAILURE;
210 }
