93 auto Attenuation::process(input_type&& sinogram) -> void {
94  if (sinogram.valid()) {
95  BOOST_LOG_TRIVIAL(debug)<< "Attenuation: Image arrived with Index: " << sinogram.index() << "to device " << sinogram.device();
96  sinograms_[sinogram.device()].push(std::move(sinogram));
97  } else {
98  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Attenuation: Received sentinel, finishing.";
99 
100  //send sentinal to processor thread and wait 'til it's finished
101  for(auto i = 0; i < numberOfDevices_; i++) {
102  sinograms_[i].push(input_type());
103  }
104 
105  for(auto i = 0; i < numberOfDevices_; i++) {
106  processorThreads_[i].join();
107  }
108  //push sentinel to results for next stage
109  results_.push(output_type());
110  BOOST_LOG_TRIVIAL(info) << "recoLib::cuda::Attenuation: Finished.";
111  }
112 }

118 auto Attenuation::processor(const int deviceID) -> void {
119  //nvtxNameOsThreadA(pthread_self(), "Attenuation");
120  CHECK(cudaSetDevice(deviceID));
121  auto avgDark_d = glados::cuda::make_device_ptr<float>(avgDark_.size());
122  auto avgReference_d = glados::cuda::make_device_ptr<float>(
123  avgReference_.size());
124  auto mask_d = glados::cuda::make_device_ptr<float>(
125  numberOfDetectors_ * numberOfProjections_);
126  CHECK(
127  cudaMemcpyAsync(avgDark_d.get(), avgDark_.data(),
128  sizeof(float) * avgDark_.size(), cudaMemcpyHostToDevice,
129  streams_[deviceID]));
130  CHECK(
131  cudaMemcpyAsync(avgReference_d.get(), avgReference_.data(),
132  sizeof(float) * avgReference_.size(), cudaMemcpyHostToDevice,
133  streams_[deviceID]));
134  //compute mask for relevant area
135  std::vector<float> mask;
136  relevantAreaMask(mask);
137  CHECK(
138  cudaMemcpyAsync(mask_d.get(), mask.data(), sizeof(float) * mask.size(),
139  cudaMemcpyHostToDevice, streams_[deviceID]));
140 
141  dim3 blocks(blockSize2D_, blockSize2D_);
142  dim3 grids(std::ceil(numberOfDetectors_ / (float)blockSize2D_),
143  std::ceil(numberOfProjections_ / (float)blockSize2D_));
144  float temp = pow(10, -5);
145  CHECK(cudaStreamSynchronize(streams_[deviceID]));
146  BOOST_LOG_TRIVIAL(info)<< "recoLib::cuda::Attenuation: Running Thread for Device " << deviceID;
147 
148  while (true) {
149  auto sinogram = sinograms_[deviceID].take();
150  if (!sinogram.valid())
151  break;
152  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Attenuation: Attenuationing image with Index " << sinogram.index();
153 
154  auto sino =
155  glados::MemoryPool<deviceManagerType>::instance()->requestMemory(
156  memoryPoolIdxs_[deviceID]);
157 
158  computeAttenuation<<<grids, blocks, 0, streams_[deviceID]>>>(
159  sinogram.container().get(), mask_d.get(), sino.container().get(),
160  avgReference_d.get(), avgDark_d.get(), temp, numberOfDetectors_,
161  numberOfProjections_, sinogram.plane());
162  CHECK(cudaPeekAtLastError());
163 
164  sino.setIdx(sinogram.index());
165  sino.setDevice(deviceID);
166  sino.setPlane(sinogram.plane());
167  sino.setStart(sinogram.start());
168 
169  //wait until work on device is finished
170  CHECK(cudaStreamSynchronize(streams_[deviceID]));
171  results_.push(std::move(sino));
172 
173  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Attenuation: Attenuationing image with Index " << sinogram.index() << " finished.";
174  }

177 auto Attenuation::init() -> void {
178  //create filter function
179  std::vector<double> filterFunction{0.5, 1.0, 1.0, 1.0, 1.5, 2.0, 3.0, 3.5, 2.0, 3.5, 3.0, 2.0, 1.5, 1.0, 1.0, 1.0, 0.5};
180  double sum = std::accumulate(filterFunction.cbegin(), filterFunction.cend(), 0.0);
181  std::transform(filterFunction.begin(), filterFunction.end(), filterFunction.begin(),
182  std::bind1st(std::multiplies<double>(), 1.0/sum));
183 
184  //read and average reference input values
185  std::vector<unsigned short> referenceValues;
186  if(pathReference_.back() != '/')
187  pathReference_.append("/");
188  std::string refPath = pathReference_ + "ref_empty_tomograph_repaired_DetModNr_";
189  readInput(refPath, referenceValues, numberOfRefFrames_);
190  //interpolate reference measurement
191  for(auto i = 0; i < numberOfRefFrames_*numberOfPlanes_; i++){
192  std::vector<int> defectDetectors(numberOfProjections_*numberOfDetectors_);
193  findDefectDetectors(referenceValues.data()+i*numberOfDetectors_*numberOfProjections_, filterFunction, defectDetectors, numberOfDetectors_, numberOfProjections_,
194  threshMin_, threshMax_);
195  interpolateDefectDetectors(referenceValues.data()+i*numberOfDetectors_*numberOfProjections_, defectDetectors, numberOfDetectors_, numberOfProjections_);
196  }
197  computeAverage(referenceValues, avgReference_);
198 
199  //read and average dark input values
200  std::vector<unsigned short> darkValues;
201  if(pathDark_.back() != '/')
202  pathDark_.append("/");
203  std::string darkPath = pathDark_ + "dark_192.168.100_DetModNr_";
204  readInput(darkPath, darkValues, numberOfDarkFrames_);
205  computeDarkAverage(darkValues, avgDark_);
206  //interpolate dark average
207  for(auto j = 0; j < numberOfPlanes_; j++){
208  for(auto i = 0; i < numberOfDetectors_; i++){
209  if(avgDark_[i + j * numberOfDetectors_] > 300.0){
210  BOOST_LOG_TRIVIAL(info) << "Interpolating dark value at detector " << i << " in plane " << j;
211  avgDark_[numberOfDetectors_ * j + i] =
212  0.5 * (avgDark_[numberOfDetectors_ * j + (i + 1)%numberOfDetectors_] +
213  avgDark_[numberOfDetectors_ * j + (i - 1)%numberOfDetectors_]);
214  }
215  }
216  }

220 auto Attenuation::computeDarkAverage(const std::vector<T>& values, std::vector<float>& average) -> void {
221  average.resize(numberOfDetectors_*numberOfPlanes_, 0.0);
222  float factor = 1.0/ (float)((float)numberOfDarkFrames_*(float)numberOfProjections_);
223  factor = 0.0;
224  for(auto i = 0; i < numberOfDarkFrames_; i++){
225  for(auto planeInd = 0; planeInd < numberOfPlanes_; planeInd++){
226  for(auto detInd = 0; detInd < numberOfDetectors_; detInd++){
227  for(auto projInd = 0; projInd < numberOfProjections_; projInd++){
228  const float val = (float)values[detInd + numberOfDetectors_*projInd + (i*numberOfPlanes_+planeInd)*numberOfDetectors_*numberOfProjections_];
229  average[detInd + planeInd*numberOfDetectors_] += val * factor;
230  }
231  }
232  }
233  }

239  average.resize(numberOfProjections_ * numberOfDetectors_ * numberOfPlanes_);
240  float factor = 1.0 / (float) numberOfRefFrames_;

275 auto Attenuation::readInput(std::string& path,
276  std::vector<T>& values, const int numberOfFrames) -> void {
277  std::vector<std::vector<T>> fileContents(numberOfDetectorModules_);
278  Timer tmr1, tmr2;
279  //if(path.back() != '/')
280  // path.append("/");
281  tmr1.start();
282  tmr2.start();
283 #pragma omp parallel for default(shared) //num_threads(9)
284  for (auto i = 1; i <= numberOfDetectorModules_; i++) {
285  std::vector<T> content;
286  //TODO: make filename and ending configurable
287  std::ifstream input(path + std::to_string(i) + ".fx", std::ios::in | std::ios::binary);
288  if (!input) {
289  BOOST_LOG_TRIVIAL(error)<< "recoLib::cuda::Attenuation: Source file " << path + std::to_string(i) + ".fx" << " could not be loaded.";
290  throw std::runtime_error("File could not be opened. Please check!");
291  }
292  //allocate memory in vector
293  std::streampos fileSize;
294  input.seekg(0, std::ios::end);
295  fileSize = input.tellg();
296  input.seekg(0, std::ios::beg);
297  content.resize(fileSize / sizeof(T));
298  input.read((char*) &content[0], fileSize);
299  fileContents[i - 1] = content;
300  }
301  tmr2.stop();
302  int numberOfDetPerModule = numberOfDetectors_ / numberOfDetectorModules_;
303  values.resize(fileContents[0].size() * numberOfDetectorModules_);
304  for (auto i = 0; i < numberOfFrames; i++) {
305  for (auto planeInd = 0; planeInd < numberOfPlanes_; planeInd++) {
306  for (auto projInd = 0; projInd < numberOfProjections_; projInd++) {
307  for (auto detModInd = 0; detModInd < numberOfDetectorModules_;
308  detModInd++) {
309  unsigned int startIndex = projInd * numberOfDetPerModule
310  + (planeInd + i * numberOfPlanes_) * numberOfDetPerModule * numberOfProjections_;
311  unsigned int indexSorted = detModInd * numberOfDetPerModule
312  + projInd * numberOfDetectors_
313  + (planeInd + i * numberOfPlanes_) * numberOfDetectors_ * numberOfProjections_;
314  std::copy(fileContents[detModInd].begin() + startIndex,
315  fileContents[detModInd].begin() + startIndex
316  + numberOfDetPerModule,
317  values.begin() + indexSorted);
318  }
319  }
320  }
321  }
322  tmr1.stop();
323  double totalFileSize = numberOfProjections_*numberOfDetectors_*numberOfPlanes_*numberOfRefFrames_*sizeof(unsigned short)/1024.0/1024.0;
324  BOOST_LOG_TRIVIAL(info)<< "recoLib::cuda::Attenuation: Reading and sorting reference input took " << tmr1.elapsed() << " s, " << totalFileSize/tmr2.elapsed() << " MByte/s.";

327 template<typename T>
328 auto Attenuation::relevantAreaMask(std::vector<T>& mask) -> void {
329  unsigned int ya, yb, yc, yd, ye;
330  unsigned int yMin, yMax;
331  double lowerLimit = (lowerLimOffset_ + sourceOffset_) / 360.0;
332  double upperLimit = (upperLimOffset_ + sourceOffset_) / 360.0;
333  //fill whole mask with ones and mask out the unrelevant parts afterwards
334  mask.resize(numberOfProjections_ * numberOfDetectors_);
335  std::fill(mask.begin(), mask.end(), 1.0);
336 
337  ya = std::round(lowerLimit * numberOfProjections_);
338  yb = ya;
339  yc = std::round(upperLimit * numberOfProjections_);
340  yd = yc;
341 
342  //slope of the straight
343  double m = ((double)ya - (double)yd) / ((double)xa_ - (double)xd_);
344 
345  ye = std::round((double)yc + ((double)xe_ - (double)xc_) * m);
346 
347  for (unsigned int x = 0; x <= xa_; x++) {
348  yMin = ya;
349  yMax = std::round(ye + m * x);
350  for (auto y = yMin; y < yMax; y++)
351  mask[x + y * numberOfDetectors_] = 0.0;
352  }
353 
354  for (auto x = xa_; x <= xc_; x++) {
355  yMin = std::round(ya + m * (x - xa_));
356  yMax = std::round(ye + m * x);
357  for (auto y = yMin; y < yMax; y++)
358  mask[x + y * numberOfDetectors_] = 0.0;
359  }
360 
361  for (auto x = xc_; x <= xd_; x++) {
362  yMin = std::round(ya + m * (x - xa_));
363  yMax = yd;
364  for (auto y = yMin; y < yMax; y++)
365  mask[x + y * numberOfDetectors_] = 0.0;
366  }
367 
368  for (auto x = xb_; x <= xf_; x++) {
369  yMin = yb;
370  yMax = std::round(yb + m * (x - xb_));
371  for (auto y = yMin; y < yMax; y++)
372  mask[x + y * numberOfDetectors_] = 0.0;
373  }
374 
375  std::fill(mask.begin(),
376  mask.begin() + lowerLimit * numberOfDetectors_ * numberOfProjections_,
377  0.0);
378  std::fill(
379  mask.begin() + upperLimit * numberOfProjections_ * numberOfDetectors_,
380  mask.end(), 0.0);
381 }

416 __global__ void computeAttenuation(
417  const unsigned short* __restrict__ sinogram_in,
418  const float* __restrict__ mask, float* __restrict__ sinogram_out,
419  const float* __restrict__ avgReference, const float* __restrict__ avgDark,
420  const float temp, const int numberOfDetectors,
421  const int numberOfProjections, const int planeId) {
422 
423  auto x = glados::cuda::getX();
424  auto y = glados::cuda::getY();
425  if (x >= numberOfDetectors || y >= numberOfProjections)
426  return;
427 
428  auto sinoIndex = numberOfDetectors * y + x;
429 
430  float numerator = (float) (sinogram_in[sinoIndex])
431  - avgDark[planeId * numberOfDetectors + x];
432 
433  float denominator = avgReference[planeId * numberOfDetectors * numberOfProjections + sinoIndex]
434  - avgDark[planeId * numberOfDetectors + x];
435 
436  if (numerator < temp)
437  numerator = temp;
438  if (denominator < temp)
439  denominator = temp;
440 
441  //comutes the attenuation and multiplies with mask for hiding the unrelevant region
442  sinogram_out[sinoIndex] = -log(numerator / denominator) * mask[sinoIndex];
443 
444 }
