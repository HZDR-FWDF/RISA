56 auto OfflineLoader::loadImage() -> glados::Image<manager_type> {
57  if (index_ >= numberOfFrames_ * numberOfPlanes_)
58  return glados::Image<manager_type>();
59  BOOST_LOG_TRIVIAL(debug) << "risa::Offlineloader: Loading image with index " << index_;
60  auto sino = glados::MemoryPool<manager_type>::instance()->requestMemory(memoryPoolIndex_);
61 #pragma omp parallel for
62  for (auto detModInd = 0; detModInd < numberOfDetectorModules_; detModInd++) {
63  ifstreams_[detModInd].get()->seekg(index_ * numberOfProjections_ * numberOfDetectorsPerModule_ * sizeof(unsigned short),
64  std::ios::beg);
65  ifstreams_[detModInd].get()->read((char*) sino.container().get()+ detModInd * numberOfDetectorsPerModule_
66  * sizeof(unsigned short) * numberOfProjections_, numberOfProjections_ * numberOfDetectorsPerModule_ * sizeof(unsigned short));
67  }
68 // ifstreams_[0].get()->seekg(index_ * numberOfProjections_ * numberOfDetectors_ * sizeof(float),
69 // std::ios::beg);
70 // ifstreams_[0].get()->read(
71 // (char*) sino.container().get(), numberOfProjections_ * numberOfDetectors_ * sizeof(float));
72  BOOST_LOG_TRIVIAL(debug)<< "Loading sinogram " << index_;
73  sino.setIdx(index_);
74  sino.setPlane(index_%numberOfPlanes_);
75  index_++;
76  sino.setStart(std::chrono::high_resolution_clock::now());
77  return sino;
78 }

105 auto OfflineLoader::readConfig(const std::string& configFile) -> bool {
106  ConfigReader configReader = ConfigReader(configFile.data());
107  int samplingRate, scanRate;
108  if (configReader.lookupValue("numberOfFanDetectors", numberOfDetectors_)
109  && configReader.lookupValue("numberOfDetectorModules",
110  numberOfDetectorModules_)
111  && configReader.lookupValue("dataInputPath", path_)
112  && configReader.lookupValue("dataFileName", fileName_)
113  && configReader.lookupValue("dataFileEnding", fileEnding_)
114  && configReader.lookupValue("numberOfPlanes", numberOfPlanes_)
115  && configReader.lookupValue("samplingRate", samplingRate)
116  && configReader.lookupValue("scanRate", scanRate)
117  && configReader.lookupValue("numberOfDataFrames", numberOfFrames_)) {
118  numberOfProjections_ = samplingRate * 1000000 / scanRate;
119  return EXIT_SUCCESS;
120  }
121 
122  return EXIT_FAILURE;
123 }
