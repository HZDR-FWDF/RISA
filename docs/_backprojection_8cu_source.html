91 auto Backprojection::process(input_type&& sinogram) -> void {
92  if (sinogram.valid()) {
93  BOOST_LOG_TRIVIAL(debug)<< "BP: Image arrived with Index: " << sinogram.index() << "to device " << sinogram.device();
94  sinograms_[sinogram.device()].push(std::move(sinogram));
95  } else {
96  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Backprojection: Received sentinel, finishing.";
97 
98  //send sentinal to processor thread and wait 'til it's finished
99  for(auto i = 0; i < numberOfDevices_; i++) {
100  sinograms_[i].push(input_type());
101  }
102  for(auto i = 0; i < numberOfDevices_; i++) {
103  processorThreads_[i].join();
104  }
105 
106  results_.push(output_type());
107  BOOST_LOG_TRIVIAL(info) << "recoLib::cuda::Backprojection: Finished.";
108  }
109 }

115 auto Backprojection::processor(const int deviceID) -> void {
116  CHECK(cudaSetDevice(deviceID));
117 
118  //init lookup tables for sin and cos
119  std::vector<float> sinLookup_h(numberOfProjections_), cosLookup_h(
120  numberOfProjections_);
121  for (auto i = 0; i < numberOfProjections_; i++) {
122  float theta = i * M_PI
123  / (float) numberOfProjections_+ rotationOffset_ / 180.0 * M_PI;
124  while (theta < 0.0) {
125  theta += 2.0 * M_PI;
126  }
127  sincosf(theta, &sinLookup_h[i], &cosLookup_h[i]);
128  }
129  CHECK(
130  cudaMemcpyToSymbol(sinLookup, sinLookup_h.data(),
131  sizeof(float) * numberOfProjections_));
132  CHECK(
133  cudaMemcpyToSymbol(cosLookup, cosLookup_h.data(),
134  sizeof(float) * numberOfProjections_));
135  //constants for kernel
136  const float scale_h = numberOfDetectors_ / (float) numberOfPixels_;
137  const float normalizationFactor_h = M_PI / numberOfProjections_ / scale_h;
138  const float imageCenter_h = (numberOfPixels_ - 1.0) * 0.5;
139  CHECK(cudaMemcpyToSymbol(normalizationFactor, &normalizationFactor_h, sizeof(float)));
140  CHECK(cudaMemcpyToSymbol(scale, &scale_h, sizeof(float)));
141  CHECK(cudaMemcpyToSymbol(imageCenter, &imageCenter_h, sizeof(float)));
142  dim3 blocks(blockSize2D_, blockSize2D_);
143  dim3 grids(std::ceil(numberOfPixels_ / (float) blockSize2D_),
144  std::ceil(numberOfPixels_ / (float) blockSize2D_));
145  if(interpolationType_ == detail::InterpolationType::linear)
146  CHECK(cudaFuncSetCacheConfig(backProjectLinear, cudaFuncCachePreferL1));
147  else if(interpolationType_ == detail::InterpolationType::neareastNeighbor)
148  CHECK(cudaFuncSetCacheConfig(backProjectNearest, cudaFuncCachePreferL1));
149  BOOST_LOG_TRIVIAL(info)<< "recoLib::cuda::BP: Running Thread for Device " << deviceID;
150  while (true) {
151  //execution is blocked until next element arrives in queue
152  auto sinogram = sinograms_[deviceID].take();
153  //if sentinel, finish thread execution
154  if (!sinogram.valid())
155  break;
156 
157  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Backprojection: Backprojecting sinogram with Index " << sinogram.index();
158 
159  //allocate device memory for reconstructed picture
160  auto recoImage =
161  glados::MemoryPool<deviceManagerType>::instance()->requestMemory(
162  memoryPoolIdxs_[deviceID]);
163 
164  if(useTextureMemory_){
165  cudaResourceDesc resDesc;
166  memset(&resDesc, 0, sizeof(resDesc));
167  resDesc.resType = cudaResourceTypeLinear;
168  resDesc.res.linear.devPtr = sinogram.container().get();
169  resDesc.res.linear.desc.f = cudaChannelFormatKindFloat;
170  resDesc.res.linear.desc.x = 32; // bits per channel
171  resDesc.res.linear.sizeInBytes = sinogram.size()*sizeof(float);
172 
173  cudaTextureDesc texDesc;
174  memset(&texDesc, 0, sizeof(texDesc));
175  texDesc.addressMode[0] = cudaAddressModeBorder;
176  texDesc.addressMode[1] = cudaAddressModeBorder;
177  texDesc.normalizedCoords = false;
178 
179  // create texture object: we only have to do this once!
180  cudaTextureObject_t tex=0;
181  CHECK(cudaCreateTextureObject(&tex, &resDesc, &texDesc, NULL));
182  backProjectTex<<<grids, blocks, 0, streams_[deviceID]>>>(tex, recoImage.container().get(),
183  numberOfPixels_, numberOfProjections_, numberOfDetectors_);
184  CHECK(cudaDestroyTextureObject(tex));
185  }else{
186  if(interpolationType_ == detail::InterpolationType::linear)
187  backProjectLinear<<<grids, blocks, 0, streams_[deviceID]>>>(
188  sinogram.container().get(), recoImage.container().get(),
189  numberOfPixels_, numberOfProjections_, numberOfDetectors_);
190  else if(interpolationType_ == detail::InterpolationType::neareastNeighbor)
191  backProjectNearest<<<grids, blocks, 0, streams_[deviceID]>>>(
192  sinogram.container().get(), recoImage.container().get(),
193  numberOfPixels_, numberOfProjections_, numberOfDetectors_);
194  }
195  CHECK(cudaPeekAtLastError());
196 
197  recoImage.setIdx(sinogram.index());
198  recoImage.setDevice(deviceID);
199  recoImage.setPlane(sinogram.plane());
200  recoImage.setStart(sinogram.start());
201 
202  //wait until work on device is finished
203  CHECK(cudaStreamSynchronize(streams_[deviceID]));
204  results_.push(std::move(recoImage));
205 
206  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Backprojection: Reconstructing sinogram with Index " << sinogram.index() << " finished.";
207  }

214  if (configReader.lookupValue("numberOfParallelProjections", numberOfProjections_)
215  && configReader.lookupValue("numberOfParallelDetectors", numberOfDetectors_)

217  && configReader.lookupValue("rotationOffset", rotationOffset_)
218  && configReader.lookupValue("blockSize2D_backProjection", blockSize2D_)
219  && configReader.lookupValue("memPoolSize_backProjection", memPoolSize_)
220  && configReader.lookupValue("interpolationType", interpolationStr)
221  && configReader.lookupValue("useTextureMemory", useTextureMemory_)
222  && configReader.lookupValue("backProjectionAngleTotal", backProjectionAngleTotal_)){
223  if(interpolationStr == "nearestNeighbour")
224  interpolationType_ = detail::InterpolationType::neareastNeighbor;
225  else if(interpolationStr == "linear")
226  interpolationType_ = detail::InterpolationType::linear;
227  else{

230  }
231 
232  return EXIT_SUCCESS;
233  }

235  return EXIT_FAILURE;

238 __global__ void backProjectLinear(const float* const __restrict__ sinogram,
239  float* __restrict__ image,
240  const int numberOfPixels,
241  const int numberOfProjections,
242  const int numberOfDetectors){
243 
244  const auto x = glados::cuda::getX();
245  const auto y = glados::cuda::getY();
246 
247  float sum = 0.0;
248 
249  if(x >= numberOfPixels || y >= numberOfPixels)
250  return;
251 
252  const int centerIndex = numberOfDetectors * 0.5;
253 
254  const float xp = (x - imageCenter[0]) * scale[0];
255  const float yp = (y - imageCenter[0]) * scale[0];
256 
257 #pragma unroll 16
258  for(auto projectionInd = 0; projectionInd < numberOfProjections; projectionInd++){
259  const float t = xp * cosLookup[projectionInd] + yp * sinLookup[projectionInd];
260  const int a = floor(t);
261  const int aCenter = a + centerIndex;
262  if(aCenter >= 0 && aCenter < numberOfDetectors){
263  sum = sum + ((float)(a + 1) - t) * sinogram[projectionInd * numberOfDetectors + aCenter];
264  }
265  if((aCenter + 1) >= 0 && (aCenter + 1) < numberOfDetectors){
266  sum = sum + (t - (float)a) * sinogram[projectionInd * numberOfDetectors + aCenter + 1];
267  }
268 
269  }
270  image[x + y * numberOfPixels] = sum * normalizationFactor[0];
271 }

274  float* __restrict__ image,
275  const int numberOfPixels,
276  const int numberOfProjections,
277  const int numberOfDetectors){
278 
279  const auto x = glados::cuda::getX();
280  const auto y = glados::cuda::getY();
281 
282  float sum = 0.0;
283 
284  if(x >= numberOfPixels || y >= numberOfPixels)
285  return;
286 
287  const float centerIndex = numberOfDetectors * 0.5;
288 
289  const float xp = (x - imageCenter[0]) * scale[0];
290  const float yp = (y - imageCenter[0]) * scale[0];
291 
292 #pragma unroll 8
293  for(auto projectionInd = 0; projectionInd < numberOfProjections; projectionInd++){
294  const int t = __fadd_rn(xp * cosLookup[projectionInd], yp * sinLookup[projectionInd] + centerIndex) ;
295  const int tCenter = t + projectionInd*numberOfDetectors;
296  if(t >= 0 && t < numberOfDetectors){
297  float val = tex1Dfetch<float>(tex, tCenter);
298  sum += val;
299  }
300  }
301  image[x + y * numberOfPixels] = sum * normalizationFactor[0];
302 }

305 __global__ void backProjectNearest(const float* const __restrict__ sinogram,
306  float* __restrict__ image, const int numberOfPixels,
307  const int numberOfProjections, const int numberOfDetectors) {
308 
309  const auto x = glados::cuda::getX();
310  const auto y = glados::cuda::getY();
311 
312  if (x >= numberOfPixels || y >= numberOfPixels)
313  return;
314 
315  float sum = 0.0;
316 
317  float *p_cosLookup = cosLookup;
318  float *p_sinLookup = sinLookup;
319 
320  const float scale = numberOfPixels / (float) numberOfDetectors;
321  const int centerIndex = numberOfDetectors * 0.5;
322 
323  const float xp = (x - (numberOfPixels - 1.0) * 0.5) / scale;
324  const float yp = (y - (numberOfPixels - 1.0) * 0.5) / scale;
325 
326 #pragma unroll 16
327  for (auto projectionInd = 0; projectionInd < numberOfProjections;
328  projectionInd++) {
329  //const int t = round(xp * cosLookup[projectionInd] + yp * sinLookup[projectionInd]) + centerIndex;
330  const int t = round(xp * *p_cosLookup + yp * *p_sinLookup) + centerIndex;
331  ++p_cosLookup; ++p_sinLookup;
332  if (t >= 0 && t < numberOfDetectors)
333  sum += sinogram[projectionInd * numberOfDetectors + t];
334  }
335  image[x + y * numberOfPixels] = sum * M_PI / numberOfProjections * scale;
336 }
