78 auto DetectorInterpolation::process(input_type&& sinogram) -> void {
79  if (sinogram.valid()) {
80  sinograms_[sinogram.device()].push(std::move(sinogram));
81  } else {
82  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::DetectorInterpolation: Received sentinel, finishing.";
83 
84  //send sentinel to all processor threads and wait 'til they're finished
85  for(auto i = 0; i < numberOfDevices_; i++){
86  sinograms_[i].push(input_type());
87  }
88 
89  //wait until all threads are finished
90  for(auto i = 0; i < numberOfDevices_; i++){
91  processorThreads_[i].join();
92  }
93 
94  //push sentinel to results for next stage
95  results_.push(output_type());
96 
97  BOOST_LOG_TRIVIAL(info)<< "recoLib::cuda::DetectorInterpolation: Finished.";
98  }
99 }

105 auto DetectorInterpolation::processor(int deviceID) -> void {
106  //nvtxNameOsThreadA(pthread_self(), "DetectorInterpolation");
107  CHECK(cudaSetDevice(deviceID));
108  BOOST_LOG_TRIVIAL(info) << "recoLib::cuda::DetectorInterpolation: Running Thread for Device " << deviceID;
109  std::vector<double> filterFunction{0.5, 1.0, 1.0, 1.0, 1.5, 2.0, 3.0, 3.5, 2.0, 3.5, 3.0, 2.0, 1.5, 1.0, 1.0, 1.0, 0.5};
110  double sum = std::accumulate(filterFunction.cbegin(), filterFunction.cend(), 0.0);
111  std::transform(filterFunction.begin(), filterFunction.end(), filterFunction.begin(),
112  std::bind1st(std::multiplies<double>(), 1.0/sum));
113  while (true) {
114  auto sinogram = sinograms_[deviceID].take();
115  if (!sinogram.valid())
116  break;
117 
118  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::DetectorInterpolation: Copy sinogram " << sinogram.index() << " to device " << deviceID;
119 
120  auto h_sino = glados::MemoryPool<glados::cuda::HostMemoryManager<unsigned short, glados::cuda::async_copy_policy>>::instance()->requestMemory(memoryPoolIdxs_[deviceID]);
121 
122  CHECK(cudaMemcpyAsync(h_sino.container().get(), sinogram.container().get(), sizeof(unsigned short)*numberOfProjections_*numberOfDetectors_, cudaMemcpyDeviceToHost, streams_[deviceID]));
123 
124  CHECK(cudaStreamSynchronize(streams_[deviceID]));
125 
126  std::vector<int> defectDetectors(numberOfDetectors_, 0);
127 
128  findDefectDetectors(h_sino.container().get(), filterFunction, defectDetectors, numberOfDetectors_, numberOfProjections_,
129  threshMin_, threshMax_);
130 
131  interpolateDefectDetectors(h_sino.container().get(), defectDetectors, numberOfDetectors_, numberOfProjections_);
132 
133  CHECK(cudaMemcpyAsync(sinogram.container().get(), h_sino.container().get(),
134  sinogram.size() * sizeof(unsigned short), cudaMemcpyHostToDevice, streams_[deviceID]));
135  CHECK(cudaStreamSynchronize(streams_[deviceID]));
136  results_.push(std::move(sinogram));
137 
138  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::DetectorInterpolation: Copy sinogram " << sinogram.index() << " to device finished.";
139  }
140 }

142 auto DetectorInterpolation::readConfig(const std::string& configFile) -> bool {
143  ConfigReader configReader = ConfigReader(configFile.data());
144  int samplingRate, scanRate;

149  && configReader.lookupValue("scanRate", scanRate)){
150  numberOfProjections_ = samplingRate * 1000000 / scanRate;
151  return EXIT_SUCCESS;
152  }
153  else
