77 auto Filter::process(input_type&& sinogram) -> void {
78  if (sinogram.valid()) {
79  BOOST_LOG_TRIVIAL(debug) << "Filter: Image arrived with Index: " << sinogram.index() << "to device " << sinogram.device();
80  sinograms_[sinogram.device()].push(std::move(sinogram));
81  } else {
82  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Filter: Received sentinel, finishing.";
83 
84  //send sentinal to processor thread and wait 'til it's finished
85  for(auto i = 0; i < numberOfDevices_; i++) {
86  sinograms_[i].push(input_type());
87  }
88 
89  for(auto i = 0; i < numberOfDevices_; i++) {
90  processorThreads_[i].join();
91  }
92  //push sentinel to results for next stage
93  results_.push(output_type());
94  BOOST_LOG_TRIVIAL(info) << "recoLib::cuda::Filter: Finished.";
95  }
96 }

102 auto Filter::processor(const int deviceID) -> void {
103  //nvtxNameOsThreadA(pthread_self(), "Filter");
104  CHECK(cudaSetDevice(deviceID));
105  auto sinoFreq = glados::cuda::make_device_ptr<cufftComplex,
106  glados::cuda::async_copy_policy>(
107  numberOfProjections_ * ((numberOfDetectors_ / 2.0) + 1));
108  dim3 dimBlock(blockSize2D_, blockSize2D_);
109  dim3 dimGrid((int) ceil((numberOfDetectors_ / 2.0 + 1) / (float) blockSize2D_),
110  (int) ceil(numberOfProjections_ / (float) blockSize2D_));
111  auto filterFunction_d = glados::cuda::make_device_ptr<float, glados::cuda::async_copy_policy>(filter_.size());
112  CHECK(cudaMemcpy(filterFunction_d.get(), filter_.data(), sizeof(float)*filter_.size(), cudaMemcpyHostToDevice));
113  BOOST_LOG_TRIVIAL(info) << "recoLib::cuda::Filter: Running Thread for Device " << deviceID;
114  while (true) {
115  auto sinogram = sinograms_[deviceID].take();
116  if (!sinogram.valid())
117  break;
118  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Filter: Filtering sinogram with Index " << sinogram.index();
119 
120  //forward transformation
121  CHECK_CUFFT(
122  cufftExecR2C(plansFwd_[deviceID],
123  (cufftReal* ) sinogram.container().get(),
124  thrust::raw_pointer_cast(&(sinoFreq[0]))));
125 
126  //Filtering
127  applyFilter<<<dimGrid, dimBlock, 0, streams_[deviceID]>>>(
128  (numberOfDetectors_ / 2) + 1, numberOfProjections_, sinoFreq.get(), filterFunction_d.get());
129 
130  CHECK(cudaPeekAtLastError());
131 
132  //reverse transformation
133  CHECK_CUFFT(
134  cufftExecC2R(plansInv_[deviceID],
135  thrust::raw_pointer_cast(sinoFreq.get()),
136  (cufftReal* ) sinogram.container().get()));
137  //wait until work on device is finished
138  CHECK(cudaStreamSynchronize(streams_[deviceID]));
139  results_.push(std::move(sinogram));
140 
141  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Filter: Filtering sinogram with Index " << sinogram.index() << " finished.";
142  }
143 }

145 auto Filter::initCuFFT(const int deviceID) -> void {
146 
147  CHECK(cudaSetDevice(deviceID));
148 
149  cudaStream_t stream;
150  cufftHandle planFwd, planInv;
151 
152  CHECK(cudaStreamCreateWithPriority(&stream, cudaStreamNonBlocking, 3));
153  streams_[deviceID] = stream;

155  CHECK_CUFFT(
156  cufftPlanMany(&planFwd, 1, &numberOfDetectors_, NULL, 0, 0, NULL, 0, 0, CUFFT_R2C, numberOfProjections_));
157 
158  CHECK_CUFFT(cufftSetStream(planFwd, stream));
159 
160  CHECK_CUFFT(
161  cufftPlanMany(&planInv, 1, &numberOfDetectors_, NULL, 0, 0, NULL, 0, 0, CUFFT_C2R, numberOfProjections_));
162 
163  CHECK_CUFFT(cufftSetStream(planInv, stream));
164 
165  plansFwd_[deviceID] = planFwd;
166  plansInv_[deviceID] = planInv;
167 }

169 auto Filter::designFilter() -> void {
170  int filterSize = numberOfDetectors_/2 + 1;
171  filter_.reserve(filterSize);
172  filter_.push_back(0.0);
173  for(auto i = 1; i < filterSize; i++){
174  //actual w at frequency axis
175  const float w = 2 * M_PI * i / (float)numberOfDetectors_;
176  if(w > M_PI*cutoffFraction_){
177  filter_.push_back(0.0);
178  continue;
179  }
180  float filterValue = 2 * i / (float)numberOfDetectors_; //* hanning(w, (float)1.0);
181  if(filterType_ == detail::FilterType::hamming)
182  filterValue *= hamming(w, cutoffFraction_);
183  else if(filterType_ == detail::FilterType::hanning)
184  filterValue *= hanning(w, cutoffFraction_);
185  else if(filterType_ == detail::FilterType::sheppLogan)
186  filterValue *= sheppLogan(w, cutoffFraction_);
187  else if(filterType_ == detail::FilterType::cosine)
188  filterValue *= cosine(w, cutoffFraction_);
189  filter_.push_back(filterValue/(float)numberOfDetectors_);
190  }
191 }

193 auto Filter::readConfig(const std::string& configFile) -> bool {
194  ConfigReader configReader = ConfigReader(configFile.data());
195  std::string filterType;
196  if (configReader.lookupValue("numberOfParallelProjections", numberOfProjections_)
197  && configReader.lookupValue("numberOfParallelDetectors", numberOfDetectors_)
198  && configReader.lookupValue("numberOfPixels", numberOfPixels_)
199  && configReader.lookupValue("blockSize2D_filter", blockSize2D_)
200  && configReader.lookupValue("filterType", filterType)
201  && configReader.lookupValue("cutoffFraction", cutoffFraction_)){
202  if(filterType == "ramp")
203  filterType_ = detail::FilterType::ramp;
204  else if(filterType == "sheppLogan")
205  filterType_ = detail::FilterType::sheppLogan;
206  else if(filterType == "hamming")
207  filterType_ = detail::FilterType::hamming;
208  else if(filterType == "hanning")
209  filterType_ = detail::FilterType::hanning;
210  else if(filterType == "cosine")
211  filterType_ = detail::FilterType::cosine;
212  else{
213  BOOST_LOG_TRIVIAL(error) << "recoLib::cuda::Filter: Requested filter mode not supported. Using Ramp-Filter.";
214  filterType_ = detail::FilterType::ramp;
215  }
216  return EXIT_SUCCESS;
217  }
218  return EXIT_FAILURE;
219 }
