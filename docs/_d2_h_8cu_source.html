83 auto D2H::process(input_type&& img) -> void {
84  if (img.valid()) {
85  if(img.index() == 0)
86  tmr_.start();
87  if((count_ % 10000) == 9999){
88  tmr_.stop();
89  reconstructionRate_ = (reconstructionRate_*(counter_-1.0) + 10000.0/(tmr_.elapsed())) / counter_;
90  counter_ += 1.0;
91  BOOST_LOG_TRIVIAL(info) << "Reconstructing at " << 10000.0/(tmr_.elapsed()) << " Images/second.";
92  tmr_.start();
93  }
94  count_++;
95  BOOST_LOG_TRIVIAL(debug)<< "Image " << img.index() << "from device " << img.device() << "arrived";
96  imgs_[img.device()].push(std::move(img));
97  } else {
98  BOOST_LOG_TRIVIAL(debug)<< "cuda::D2H: Received sentinel, finishing.";
99 
100  //send sentinal to processor threads and wait 'til they're finished
101  for(auto i = 0; i < numberOfDevices_; i++) {
102  imgs_[i].push(input_type());
103  }
104 
105  for(auto i = 0; i < numberOfDevices_; i++) {
106  processorThreads_[i].join();
107  }
108 
109  //push sentinel to results for next stage
110  results_.push(output_type());
111  BOOST_LOG_TRIVIAL(info) << "cuda::D2H: Finished.";
112  }
113 }

119 auto D2H::processor(const int deviceID) -> void {
120  //nvtxNameOsThreadA(pthread_self(), "D2H");
121  CHECK(cudaSetDevice(deviceID));
122  BOOST_LOG_TRIVIAL(info) << "recoLib::cuda::D2H: Running Thread for Device " << deviceID;
123  while (true) {
124  auto img = imgs_[deviceID].take();
125  if (!img.valid()) {
126  BOOST_LOG_TRIVIAL(debug)<< "invalid image arrived.";
127  break;
128  }
129 
130  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::D2H: Copy sinogram " << img.index() << " from device " << img.device();
131 
132  //copy image from device to host
133  auto ret = glados::MemoryPool<hostManagerType>::instance()->requestMemory(
134  memoryPoolIdx_);
135  CHECK(
136  cudaMemcpyAsync(ret.container().get(), img.container().get(),
137  img.size() * sizeof(float), cudaMemcpyDeviceToHost, streams_[deviceID]));
138  ret.setIdx(img.index());
139  ret.setPlane(img.plane());
140  ret.setStart(img.start());
141  CHECK(cudaStreamSynchronize(streams_[deviceID]));
142 
143  //wait until work on device is finished
144  results_.push(std::move(ret));
145 
146  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::D2H: Copy sinogram " << img.index() << " from device " << img.device() << " finished.";
147  }
148 }
